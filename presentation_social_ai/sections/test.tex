       \begin{column}{0.5\hsize}
       $$ \small v(k+1) = v(k) + dt \cdot a(k) $$
       \end{column}
   \end{columns}

   \hfil

   \hfil

   Acceleration Equations:
   \begin{columns}[c]
       \begin{column}{0.5\hsize}
           \centering
           $$a(k) = \frac{2}{dt^2} \Bigl( s(k+1) - s(k) - dt \cdot v(k) \Bigr)$$
       \end{column}

       \begin{column}{0.5\hsize}
           \centering
           $$ a(k) = \frac{1}{dt} \Bigl( v(k+1) - v(k) \Bigr)$$
       \end{column}
   \end{columns}
